var ancho : int
var largo : int
var x : float
var y : float


{
  ancho = 10
  largo = 5
  y = 10
  x = y
  y = ancho + largo / x

  var z : float = (ancho + largo) / y

  si (ancho > largo) {
    y = x + 5
  }

  si (largo < ancho) {
    y = x + z
  } sino {
    y = x + z + z
  }

  mientras (largo > ancho) {
    ancho = ancho + 1
  }

  mientras (ancho > largo) {
    largo = largo + 1
  }
}
