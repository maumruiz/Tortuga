programa fuciones

func cuadrado() {
  var lados : int = 4
  var perimetro : int
  perimetro = 4 * lados
}

func perimTriangulo(largo:float) {
  var area : float
  repetir(3) {
    area = area + largo
  }
}

func poligono(n:int, largo:float) {
  var lados : int = n
  var perimetro : float
  repetir(n) {
    perimetro = perimetro + largo
  }
}

var triangulo : float = 3.5
var n : int = 6

cuadrado()
perimTriangulo(triangulo)
poligono(n, 5.2)
