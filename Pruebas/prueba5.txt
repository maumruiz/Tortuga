programa poligonos

func uno(n: int, m: int, o: int) {
  mientras(n > m) {
      n = m * o
      o = n / m + m
  }
}

func dos(n: int, m: int, o: int) {
    var a : int = 3
    var b : int = 6
    var c : int = 9

    n = a + m * o + (a * b + c)
    m = b - a + c
}


{
    var a : int = 3
    var b : int = 6
    var c : int = 9
    uno(a, b, c)
    dos(a, b, c)
}
